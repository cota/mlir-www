4 """Models DAGs of scalar math expressions.
5 
6 Used for generating region bodies at the "math" level where they are still type
7 polymorphic. This is modeled to be polymorphic by attribute name for interop
8 with serialization schemes that are just plain-old-dicts.
9 
10 These classes are typically not user accessed and are created as a by-product
11 of interpreting a comprehension DSL and model the operations to perform in the
12 op body. The class hierarchy is laid out to map well to a form of YAML that
13 can be easily consumed from the C++ side, not necessarily for ergonomics.
14 """

72 class ScalarExpression(YAMLObject):
73  """An expression on scalar values.
74 
75  Can be one of:
76  - ScalarApplyFn
77  - ScalarArg
78  - ScalarSymbolicCast
79  """
80  yaml_tag = "!ScalarExpression"
81 
82  def __init__(self,
83  scalar_apply: Optional[ScalarApplyFn] = None,
84  scalar_arg: Optional[ScalarArg] = None,
85  symbolic_cast: Optional[ScalarSymbolicCast] = None):
86  if (bool(scalar_apply) + bool(scalar_arg) + bool(symbolic_cast)) != 1:
87  raise ValueError(
88  "One of 'scalar_apply', 'scalar_block_arg', 'symbolic_cast' must be "
89  "specified")
90  self.scalar_apply = scalar_apply
91  self.scalar_arg = scalar_arg
92  self.symbolic_cast = symbolic_cast
93 
94  def to_yaml_custom_dict(self):
95  if self.scalar_apply:
96  return dict(scalar_apply=dict(
97  fn_name=self.scalar_apply.fn_name,
98  operands=list(self.scalar_apply.operands),
99  ))
100  elif self.scalar_arg:
101  return dict(scalar_arg=self.scalar_arg.arg)
102  elif self.symbolic_cast:
103  # Note that even though operands must be arity 1, we write it the
104  # same way as for apply because it allows handling code to be more
105  # generic vs having a special form.
106  return dict(symbolic_cast=dict(type_var=self.symbolic_cast.to_type.name,
107  operands=[self.symbolic_cast.operand]))
108  else:
109  raise ValueError(f"Unexpected ScalarExpression type: {self}")
