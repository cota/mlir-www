42 /// This class defines the main interface for locations in MLIR and acts as a
43 /// non-nullable wrapper around a LocationAttr.
44 class Location {
45 public:
46  Location(LocationAttr loc) : impl(loc) {
47  assert(loc && "location should never be null.");
48  }
49  Location(const LocationAttr::ImplType *impl) : impl(impl) {
50  assert(impl && "location should never be null.");
51  }
52 
53  /// Return the context this location is uniqued in.
54  MLIRContext *getContext() const { return impl.getContext(); }
55 
56  /// Access the impl location attribute.
57  operator LocationAttr() const { return impl; }
58  LocationAttr *operator->() const { return const_cast<LocationAttr *>(&impl); }
59 
60  /// Type casting utilities on the underlying location.
61  template <typename U> bool isa() const { return impl.isa<U>(); }
62  template <typename U> U dyn_cast() const { return impl.dyn_cast<U>(); }
63  template <typename U> U cast() const { return impl.cast<U>(); }
64 
65  /// Comparison operators.
66  bool operator==(Location rhs) const { return impl == rhs.impl; }
67  bool operator!=(Location rhs) const { return !(*this == rhs); }
68 
69  /// Print the location.
70  void print(raw_ostream &os) const { impl.print(os); }
71  void dump() const { impl.dump(); }
72 
73  friend ::llvm::hash_code hash_value(Location arg);
74 
75  /// Methods for supporting PointerLikeTypeTraits.
76  const void *getAsOpaquePointer() const { return impl.getAsOpaquePointer(); }
77  static Location getFromOpaquePointer(const void *pointer) {
78  return LocationAttr(reinterpret_cast<const AttributeStorage *>(pointer));
79  }
80 
81 protected:
82  /// The internal backing location attribute.
83  LocationAttr impl;
84 };
85 
86 inline raw_ostream &operator<<(raw_ostream &os, const Location &loc) {
87  loc.print(os);
88  return os;
89 }
