38 // Extract an LLVM IR type from the LLVM IR dialect type.
39 static Type unwrap(Type type) {
40  if (!type)
41  return nullptr;
42  auto *mlirContext = type.getContext();
43  if (!LLVM::isCompatibleType(type))
44  emitError(UnknownLoc::get(mlirContext),
45  "conversion resulted in a non-LLVM type");
46  return type;
47 }

49 static Optional<Type>
50 convertScalableVectorTypeToLLVM(ScalableVectorType svType,
51  LLVMTypeConverter &converter) {
52  auto elementType = unwrap(converter.convertType(svType.getElementType()));
53  if (!elementType)
54  return {};
55 
56  auto sVectorType =
57  LLVM::LLVMScalableVectorType::get(elementType, svType.getShape().back());
58  return sVectorType;
59 }

61 template <typename OpTy>
62 class ForwardOperands : public OpConversionPattern<OpTy> {
63  using OpConversionPattern<OpTy>::OpConversionPattern;
64 
65  LogicalResult
66  matchAndRewrite(OpTy op, ArrayRef<Value> operands,
67  ConversionPatternRewriter &rewriter) const final {
68  if (ValueRange(operands).getTypes() == op->getOperands().getTypes())
69  return rewriter.notifyMatchFailure(op, "operand types already match");
70 
71  rewriter.updateRootInPlace(op, [&]() { op->setOperands(operands); });
72  return success();
73  }
74 };

98 /// Populate the given list with patterns that convert from ArmSVE to LLVM.
99 void mlir::populateArmSVEToLLVMConversionPatterns(LLVMTypeConverter &converter,
100  RewritePatternSet &patterns) {
101  converter.addConversion([&converter](ScalableVectorType svType) {
102  return convertScalableVectorTypeToLLVM(svType, converter);
103  });
104  converter.addSourceMaterialization(addUnrealizedCast);
105 
106  // clang-format off
107  patterns.add<ForwardOperands<CallOp>,
108  ForwardOperands<CallIndirectOp>,
109  ForwardOperands<ReturnOp>>(converter,
110  &converter.getContext());
111  patterns.add<SdotOpLowering,
112  SmmlaOpLowering,
113  UdotOpLowering,
114  UmmlaOpLowering,
115  VectorScaleOpLowering>(converter);
116  // clang-format on
117 }
